// ALTERAÇÃO PRINCIPAL: suporte a duas overlays (normal + deslocada 100px para cima) // 👉 Adicione isso no topo das configs const CONFIG = { BOTTOM_MARGIN: 120, GAP_BETWEEN_BOX_AND_TEXT: 36, VARIANT_OFFSET_Y: 100 // novo }; // ===== NOVO: duas overlays ===== const OVERLAY_DEFAULT = 'overlay.png'; const OVERLAY_ALT = 'overlay-alt.png'; // nova overlay (mesmo tamanho 1080x1350) let overlayDefaultImg = null; let overlayAltImg = null; // ===== ALTERAÇÃO NO BOOT ===== (async function boot(){ overlayDefaultImg = await loadImage(OVERLAY_DEFAULT); overlayAltImg = await loadImage(OVERLAY_ALT); scheduleRender(); })(); // ===== ALTERAÇÃO NO RENDER ===== async function render(variant = 'default'){ await ensureFonts(); const W = 1080, H = 1350; const dpr = window.devicePixelRatio || 1; canvas.width = W * dpr; canvas.height = H * dpr; ctx = canvas.getContext('2d'); ctx.setTransform(dpr, 0, 0, dpr, 0, 0); ctx.clearRect(0,0,W,H); // Fundo if(bgImg){ if(!bgState.dw || !bgState.dh){ centerAndClampBg(W, H); } ctx.drawImage(bgImg, bgState.x, bgState.y, bgState.dw, bgState.dh); } else { ctx.fillStyle = '#333'; ctx.fillRect(0,0,W,H); } // Overlay const overlayToUse = variant === 'alt' ? overlayAltImg : overlayDefaultImg; if (overlayToUse) ctx.drawImage(overlayToUse, 0, 0, W, H); // ===== OFFSET NOVO ===== const OFFSET_Y = variant === 'alt' ? CONFIG.VARIANT_OFFSET_Y : 0; // ===== TEXTOS ===== const headline = (els.manchete.value || ''); const HEADLINE_FONT_SIZE = 64; const HEADLINE_LEFT_PADDING = 80; const HEADLINE_RIGHT_PADDING = 80; const LINE_HEIGHT = Math.round(HEADLINE_FONT_SIZE * 1.24); ctx.fillStyle = '#fff'; ctx.font = `700 italic ${HEADLINE_FONT_SIZE}px Roboto`; const maxTextW = W - HEADLINE_LEFT_PADDING - HEADLINE_RIGHT_PADDING; const lines = wrapWithNewlinesNoBlank(headline, maxTextW).slice(0,6); const L = Math.max(1, lines.length); let yFirstBaseline = H - CONFIG.BOTTOM_MARGIN - (L - 1) * LINE_HEIGHT; // 👇 AQUI A MÁGICA yFirstBaseline -= OFFSET_Y; // Badge let badgeText = (els.destaque.value || '').toUpperCase(); const BADGE_FONT_SIZE = 64; const BADGE_HEIGHT = 80; const BADGE_LEFT_PADDING = 80; ctx.font = `700 italic ${HEADLINE_FONT_SIZE}px Roboto`; const mHeadline = ctx.measureText(lines[0] || 'Ag'); const ascentHeadline = mHeadline.actualBoundingBoxAscent || 50; const firstLineTopY = yFirstBaseline - ascentHeadline; const badgeY = firstLineTopY - CONFIG.GAP_BETWEEN_BOX_AND_TEXT - BADGE_HEIGHT; ctx.fillStyle = '#FFD400'; ctx.fillRect(0, badgeY, 400, BADGE_HEIGHT); ctx.fillStyle = '#111'; ctx.font = `400 ${BADGE_FONT_SIZE}px "Bebas Neue"`; ctx.fillText(badgeText, BADGE_LEFT_PADDING, badgeY + 60); // Texto ctx.fillStyle = '#fff'; ctx.font = `700 italic ${HEADLINE_FONT_SIZE}px Roboto`; let y = yFirstBaseline; for(const ln of lines){ ctx.fillText(ln, HEADLINE_LEFT_PADDING, y); y += LINE_HEIGHT; } els.download.disabled = false; } // ===== NOVO BOTÃO: gerar as duas ===== function gerarDuasImagens(){ // versão normal render('default'); setTimeout(()=>{ const img1 = canvas.toDataURL('image/png'); downloadImage(img1, 'noticia-IG.png'); // versão deslocada render('alt'); setTimeout(()=>{ const img2 = canvas.toDataURL('image/png'); downloadImage(img2, 'noticia-FB.png'); }, 100); }, 100); } function downloadImage(data, name){ const a = document.createElement('a'); a.href = data; a.download = name; a.click(); } // botão const btn = document.createElement('button'); btn.textContent = 'Baixar 2 versões'; btn.onclick = gerarDuasImagens; document.querySelector('.btns').appendChild(btn);
